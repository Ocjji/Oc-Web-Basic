div.addEventListener('click', () => {
            // offset : 0 ~ 1 (0 ~ 100%)
            //'3초' 처음과 끝은 생략 가능하다
            div.animate([
                { transform: "translateX(0px)", offset: 0 },
                { transform: "translateX(100px)", offset: 0.05 },
                { transform: "translateX(200px)", offset: 0.1 },
                { transform: "translateX(300px)", offset: 0.2 },
                { transform: "translateX(400px)", offset: 0.5 },
                { transform: "translateX(500px)", offset: 1 },
            ], 3000);
        });
