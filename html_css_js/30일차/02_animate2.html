div.addEventListener('click', () => {
            // alert('클릭!');
            div.animate([
                { transform: "rotate(0) scale(1)", opacity: 0 },
                { transform: "rotate(360deg) scale(3)", opacity: 1 },
                { transform: "rotate(720deg) scale(1)", opacity: 0 },
            ], { duration: 3000, iterations: Infinity });
        });
