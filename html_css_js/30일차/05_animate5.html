div.addEventListener('click', () => {
            easing을 통해 효과 속성 적용
            div.animate([
                { transform: "translateX(0px)", easing: 'ease' },
                { transform: "translateX(300px)", easing: 'ease-in' },
                { transform: "translateX(600px)", easing: 'linear' },
            ], 3000);
        });
